རྐུན་མ།

	ང་ཨ་རཱོན་ལ་ཐུག་སྐབས། ང་ནི་ད་དུང་ཡང་རྐུན་མ་ཞིག་ཡིན། ངའི་ལོ་ན་བཅོ་ལྔ་ཡིན་ཡང་། ང་ནི་ཉམས་མྱོང་ལྡན་ཞིང་ལམ་ལྷོངས་ཅན་གྱི་རྐུན་མ་ཞིག་ཡིན། 

	ང་ཨ་རཱོན་གྱི་སར་བཅར་དུས། ཁོས་གྱད་རྩལ་འགྲན་སྡུར་ཞིག་ལ་ལྟ་ཡི་འདུག ཁོའི་ལོ་ནི་ཧ་ལམ་ཉི་ཤུ་ཙམ་དང་། གཟུགས་པོ་རིང་ཞིང་སྐམ་པ་དང་། ངའི་དམིགས་ཡུལ་དང་འཚམ་པའི་སེམས་པ་བཟང་ཞིང་བབ་ཆགས་ཤིག་རེད། ཉེ་བའི་ཆར་རྐུན་མ་རྒྱག་རྒྱུ་ལམ་ལྷོངས་དེ་ཙམ་མ་བྱུང་བས། ཕལ་ཆེར་ན་གཞོན་དེར་རྐུན་མ་རྒྱག་ཐུབ་ཀྱི་རེད་སྙམ། ཁོ་གྱད་རྩལ་འགྲན་སྡུར་ལ་ངོ་མཚར་ཆེན་པོ་སྐྱེས་འདུག་པར་སྣང་། ལུས་ལ་སྣུམ་ཤད་བྱས་པའི་སྐྱེས་པ་གཉིས་ཀྱིས་འདམ་བག་ནང་དུ་འཇུ་རེས་དང་། སྣ་ཤུ་རྒྱག་བཞིན་བླ་ཤ་ལ་གཞུ་ཡི་འདུག ངས་ཨ་རཱོན་ལ་ཁ་བརྡ་བྱས་སྐབས། ཁོས་ང་ནི་མི་ཆ་མེད་ཅིག་ཡིན་པ་ཡང་ཤེས་མེད་པར་སྣང་། 

	ངས། “ཁྱེད་རང་རང་གྱད་པ་ཞིག་དང་འདྲ་བོ་འདུག” ཅེས་ལབ་པ་ཡིན། 

	ཁོས། “ཁྱེད་རང་ཡང་གྱད་པ་འདྲ་བོ་འདུག” ཅེས་བཤད་པ་དེས་ང་ཡུད་ཙམ་རིང་རང་གི་དོ་སྣང་ལས་ཕྱིར་དྲངས། རྒྱུ་མཚན་གང་ལགས་ཤེ་ན། སྐབས་དེར་ང་ནི་ལུས་སྐམ་ཞིང་རིད་པས། ཕྱི་གཟུགས་ཀྱི་ཆ་ནས་སྟོབས་ཆེན་པོ་མིན། 

	ངས། “ཡོད། ངས་མཚམས་མཚམས་སུ་གྱད་རྩལ་འགྲན་གྱི་ཡོད།” ཅེས་ལབ་པ་ཡིན། 

	“ཁྱེད་ཀྱི་མིང་ལ་ཅི་ཟེར།”

	ངས། “ཏི་པག་(Deepak)ཟེར་གྱི་ཡོད།” ཅེས་རྫུན་བཤད་པ་ཡིན། 

	ཏི་པག་ནི་ངའི་མིང་ལྔ་བ་དེ་ཡིན། ངས་དེ་སྔ་རང་གི་མིང་ལ་རན་བྷིར་དང་། སུ་ངྷིར། ཐི་ལོཀ སུ་རེན་ཏིར་བཅས་བོས་བ་ཡིན། 

	ཁ་བརྡའི་མགོ་རུ་སྐད་ཆ་དེ་དག་བཤད་རྗེས། ཨ་རཱོན་གྱིས་གྱད་རྩལ་འགྲན་སྡུར་སྐོར་མང་པོ་ཤོད་ཀྱི་འདུག་པས། ང་ལ་སྐད་ཆ་མང་པོ་ཤོད་རྒྱུ་མི་འདུག ཏོག་ཙམ་རྗེས་སུ་ཁོ་ལྟད་མོ་བའི་ཁྲོད་ནས་བུད་སོང་བས། ང་ཡང་ཁོའི་རྗེས་སུ་ཕྱིན་པ་ཡིན། 

	ཁོས། “ཁྱེད་རང་སྐྱིད་པོ་འདུག་གམ།” ཞེས་དྲིས། 

	ངས་ཁོ་ལ་ཆེས་ཡིད་དུ་འོང་བའི་འཛུམ་མདངས་ཤིག་བསྟན་ནས། “ངས་ཁྱེད་ཀྱི་ཆེད་དུ་ལས་ཀ་བྱེད་འདོད་ཡོད།” ཅེས་བཤད། 

	ཁོས་འགྲོ་མཚམས་མ་བཞག་པར། “ཅི་ཞིག་གིས་ང་ལ་ལས་ཀ་བ་ཞིག་དགོས་ཀྱི་རེད་བསམ་བཅུག་བྱུང་ངམ།” 

	ངས། “ངས་ཉིན་གང་པོར་ཉུལ་ནས། ལས་ཀ་སྒྲུབ་ས་ཡག་ཤོས་དེ་འཚོལ་བཞིན་ཡོད། ངས་ཁྱེད་རང་མཐོང་བ་དང་། ཁྱེད་ལས་ལྷག་པའི་མི་ཞིག་མེད་པ་ཚོར་བྱུང་།” ཞེས་བཤད་པ་ཡིན། 

	ཁོས། “ཁྱེད་ཀྱིས་ང་ལ་སྤྲེལ་ལད་འཁྲབ་ཀྱི་འདུག” ཅེས་བཤད། 

	“དེ་འདྲ་རང་མ་རེད།” 

	“གང་ལྟར། ཁྱེད་ཀྱིས་ངའི་ལས་མི་བྱེད་ཐུབ་ཀྱི་མ་རེད།” 

	“ག་རེ་བྱས་ནས།”

	“ངས་ཁྱེད་ལ་གླ་ཆ་སྤྲོད་ཐུབ་ཀྱི་མེད། ” 

	ངས་སྐར་མ་གཅིག་གི་རིང་ལ་དེའི་སྐོར་བསམ་བློ་བཏང་བ་ཡིན། ངས་ཕལ་ཆེར་མི་དེར་ལྟར་ནོར་ཤོར་བ་འདྲ།  

	ངས། “ཁྱེད་ཀྱིས་ང་གསོ་ཐུབ་བམ།” ཞེས་དྲིས། 

	ཁོས། “ཁྱེད་ཀྱིས་ཁ་ཟས་གཡོས་སྦྱོར་བྱེད་ཐུབ་བམ།” ཞེས་ཕྱིར་དྲིས། 

	ངས། “ཁ་ཟས་གཡོས་སྦྱོར་བྱེད་ཐུབ།” ཅེས་རྫུན་བཤད་པ་ཡིན། 

	ཁོས། “གལ་ཏེ། ཁྱེད་ཀྱིས་ཁ་ཟས་གཡོས་སྦྱོར་བྱེད་ཐུབ་ན། ངས་ཁྱེད་རང་གསོ་ཆོག” ཅེས་བཤད། 

	ཁོས་ང་ཁང་པའི་ནང་དུ་ཁྲིད་ནས་བར་ཁྱམས་སུ་ཉལ་ཆོག་ཟེར། འོན་ཀྱང་། མཚན་མོ་དེར་ཏོག་ཙམ་མ་གཏོགས་ང་སླར་ཡང་སྲང་ལམ་འགྲོ་དགོས་ཐུག་བྱུང་། ངས་གཡོས་སྦྱོར་བྱས་པའི་ཁ་ལག་དེ་ཞིམ་པོ་གཏན་ནས་མེད་པ་འདྲ། རྒྱུ་མཚན་གང་ལགས་ཤེ་ན། ཨ་རཱོན་གྱིས་ཁ་ལག་དེ་ཁྱིམ་མཚེས་ཀྱི་ཞི་མི་ལ་སྤྲད་ནས་ང་ལ་སོང་ཟེར། ཡིན་ན་ཡང་། ངས་གང་ཐུབ་ཅི་ཐུབ་ཀྱིས་འཛུམ་མདངས་བསྟན་ནས་ཡར་འགྲོ་མར་འགྲོ་བྱས་པས། ཁོ་གད་མོ་མི་ཤོར་རང་ཤོར་ཆགས། ཁོ་ཉལ་ཁྲིའི་སྟེང་དུ་བསྡད་དེ།  སྐར་མ་ལྔའི་རིང་ལ་གད་མོ་ཡག་ཐག་ཆོད་ཅིག་བགད་མཐར། ངའི་མགོ་ལ་བྱིལ་བྱིལ་བྱས་ནས། གལ་ཡོད་པ་མ་རེད། ཞོགས་པ་ངས་ཁྱེད་ལ་ཁ་ལག་གཡོས་སྦྱོར་བྱེད་སྟངས་བསླབས་ཆོག་ཟེར། 

	ཁོས་ང་ལ་ཁ་ལག་བཟོ་སྟངས་བསླབས་པ་མ་ཟད། ང་དང་ཁོའི་མིང་འབྲི་སྟངས་ཀྱང་བསླབས། མ་ཟད་མི་རིང་བར་ང་ལ་ཚིག་གྲུབ་ཧྲིལ་བོ་འབྲི་སྟངས་དང་། སྦ་ཁུག་ནང་ལ་སྒོར་མོ་མེད་ཚེ་ཤོག་བུའི་སྒང་ལ་ཐོ་རྒྱག་སྟངས་བསླབས། 

	ཨ་རཱོན་ལ་ལས་ཀ་བྱེད་རྒྱུ་ནི་སྤྲོ་སྣང་ལྡན་པའི་བྱ་བ་ཞིག་རེད། ཞོགས་པ་ཇ་བསྐོལ་བ་དང་། དེའི་རྗེས་སུ་ཉོ་ཅ་རྒྱག་ཏུ་ཕྱིན་པ་ཡིན། ངས་ཉིན་ཞིག་གི་མཁོ་ཆས་དེ་ཉོ་བ་དང། ཉིན་རེར་པད་ཤག་༢༥་ཁེ་བསང་བསགས། ངས་ཨ་རཱོན་ལ་འབྲས་རྒྱ་མ་གང་གི་རིན་གོང་ནི་པད་ཤག་༥༦་(དངོས་ཡོད་དེ་འདྲ་རེད)ཅེས་ལབ་ཀྱི་ཡོད་ཀྱང་། ང་ལ་པད་ཤག་༥༠་ལ་རག་གི་ཡོད། ཁོས་ངས་འདི་ལྟར་སྒོར་མོ་ཏོག་ཙམ་གསོག་གི་ཡོད་པ་ཤེས་ཀྱི་ཡོད་རྒྱུ་རེད་བསམས། ཡིན་ན་ཡང་། ཁོས་གལ་ཆེན་པོར་མ་བརྩིས། ཁོས་ང་ལ་རྒྱུན་ལྡན་གྱི་གླ་ཆ་ཞིག་སྤྲོད་ཀྱི་མེད།
